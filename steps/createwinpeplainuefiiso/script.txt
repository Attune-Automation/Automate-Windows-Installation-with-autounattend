if {kickstartedBootLoaderIsUefi}
then
        
    if uname | \
        grep -iq "darwin"
    then
        echo "is macOS, skipping"
        
    else
        echo "not macOS"
        
        cd "{automationWorkerBaseDirectory}/build-winpe-iso"
        
        bytes=$(du --summarize winpe_staging | awk '{print $1}')
        ((bytes=bytes+10000))
        
        ISO="{automationWorkerBaseDirectory}/winpe_plain_uefi.iso"
        mkfs.msdos -v -C "${ISO}" "${bytes}"
    fi
    
else
    echo "Skipping for BIOS boot."
fi
