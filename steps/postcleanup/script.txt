DIR="{automationWorkerBaseDirectory}/build-{newOsNode.fqn}"
if [ -d "${DIR}" ]
then
    rm -rf "${DIR}" && echo "Removed ${DIR}"
else
    echo "No build directory to remove."
fi

DIR="{automationWorkerBaseDirectory}/drivers-{newOsNode.fqn}"
if [ -d "${DIR}" ]
then
    rm -rf "${DIR}" && echo "Removed ${DIR}"
else
    echo "No drop in drivers directory to remove."
fi

# The wildcard will allow removal of 
# - kickstart_{newOsNode.fqn}.iso
# - drivers_{newOsNode.fqn}.iso
ISO={automationWorkerBaseDirectory}/*{newOsNode.fqn}.iso

if [ -z "${ISO}" ] 
then
    echo "No ISO to remove."
else
    rm -fv ${ISO}
fi
