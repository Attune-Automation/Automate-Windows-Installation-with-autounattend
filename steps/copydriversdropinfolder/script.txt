cd {automationWorkerBaseDirectory}/build-{newOsNode.fqn}

DRIVERS_DROP_IN_FOLDER="{automationWorkerBaseDirectory}/drivers-{newOsNode.fqn}"

        "${DESTINATION_DRIVERS_FOLDER}"
else

fi
