cd "{automationWorkerBaseDirectory}/build-winpe-iso"
FOLDER="winimg"
mkdir -p "${FOLDER}"

# Case insensitve globbing
shopt -s nocaseglob

ISO=$(ls *SERVER*.iso)

if uname | \
    grep -iq "darwin"
then
    echo "is macOS"
    
    #hdiutil attach "${ISO}" -mountroot "${FOLDER}"
    hdiutil attach "${ISO}" -mountpoint "${FOLDER}"    
else
    echo "not macOS"
    
    mount --options "uid={automationWorkerLinuxUser.user}" \
        --options "gid={automationWorkerLinuxUser.user}" \
        "${ISO}" "${FOLDER}"
fi
