cd "{automationWorkerLinuxBaseDirectory}/build-winpe-iso"

if {kickstartedBootLoaderIsBios} 
then
    ISO_NAME="winpe_plain_bios.iso"
    
    if [[ "$(uname)" == "Darwin" ]]
    then
        
        export PATH="${HOME}/Parallels/homebrew/bin:$PATH"
        echo "Using $(which xorriso)"

        xorriso -as mkisofs \
            -b boot.img \
            -no-emul-boot \
            -c BOOT.CAT \
            -iso-level 3 \
            -udf \
            -joliet-long \
            -relaxed-filenames \
            -allow-limited-size \
            -J \
            -full-iso9660-filenames \
            -disable-deep-relocation \
            -omit-version-number \
            -V "KS_WIN" \
            -o "{automationWorkerLinuxBaseDirectory}/${ISO_NAME}" \
            winpe_staging
            
    elif [[ -f /etc/redhat-release ]] || [[ -f /etc/lsb-release ]] || [[ -f /etc/debian_version ]]
    then
    
        mkisofs \
            -b boot.img \
            -no-emul-boot \
            -c BOOT.CAT \
            -iso-level 3 \
            -udf \
            -joliet-long \
            -relaxed-filenames \
            -allow-limited-size \
            -J \
            -full-iso9660-filenames \
            -disable-deep-relocation \
            -omit-version-number \
            -V "KS_WIN" \
            -o "{automationWorkerLinuxBaseDirectory}/${ISO_NAME}" \
            winpe_staging
    
    else
    
        echo "Unsupported operating system."
    fi
        
else
    echo "Skipping for UEFI ISO creation."
fi
